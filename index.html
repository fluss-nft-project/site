    <script type="text/javascript">
        var ADDR = null;
        var SIGNER = null;
        var PROVIDER = new ethers.providers.AlchemyProvider(NETWORK, ALCHEMY_PID);
        var CONTRACT = new ethers.Contract(CONTRACT_ADDR, CONTRACT_ABI, PROVIDER);
        var NUM_SOLD = 0;
        var LABEL_CHOICES = ["SOY IT", "RUG ME", "SIMP"];
        

        // doc ready check
        window.addEventListener('load', async () => {
            document.getElementById("minus").addEventListener("click", function() {
                var value = parseInt(document.getElementById("number").innerText, 10)%100 - 1;
                if (value <= 0) {
                    value = 1;
                }
                document.getElementById("number").innerText = value;
            });

            document.getElementById("plus").addEventListener("click", function() {
                var value = parseInt(document.getElementById("number").innerText, 10);
                console.log("value", value);
                document.getElementById("number").innerText = value%100+1;
            });

            async function refreshInfo() {
                var NUM_SOLD = await CONTRACT.totalSupply();

                document.getElementById("sold").innerHTML = NUM_SOLD+"/5000 SOLD";
            }
            refreshInfo();
            setInterval(refreshInfo, REFRESH_FREQ);

            if (ADDR == null) {
                document.getElementById("connect").innerHTML = "Not connected";
                document.getElementById("mint-button").innerHTML = "CONNECT WALLET";
            }

            document.getElementById("mint-button").addEventListener("click", async function() {
                if (typeof window.ethereum == 'undefined') {
                    alert("Please install MetaMask to use this site.");
                    return;
                }

                if (SIGNER == null || ADDR == null) {
                    var accs = await ethereum.request({ method: 'eth_requestAccounts' });
                    ADDR = accs[0];
                    console.log("addr", ADDR);

                    PROVIDER = new ethers.providers.Web3Provider(window.ethereum);
                    CONTRACT = new ethers.Contract(CONTRACT_ADDR, CONTRACT_ABI, PROVIDER);
                    if (ADDR.includes(".eth")) {
                        ADDR = PROVIDER.resolveName(ADDR);
                    }
                    SIGNER = PROVIDER.getSigner();
                    document.getElementById("connect").innerHTML = "Connected as " + ADDR.substring(0,3) + "..." + ADDR.substring(ADDR.length-3, ADDR.length);
                    document.getElementById("mint-button").innerHTML = LABEL_CHOICES[Math.floor(Math.random() * LABEL_CHOICES.length)];

                    return;
                }

                var quantity = parseInt(document.getElementById("number").innerText, 10);
                var MINT_FEE = await CONTRACT.connect(SIGNER).computePrice(quantity);
                document.getElementById("minter").classList.add("minting");

                CONTRACT.connect(SIGNER).mint(quantity, {value: MINT_FEE}).then(function(txn) {
                    txn.wait(NUM_CONFIRMATION).then(async function(receipt) {
                        console.log(receipt);
                        var _NETWORK = NETWORK;
                        if (NETWORK == "mainnet") {
                            _NETWORK = ""
                        } else {
                            _NETWORK += ".";
                        }

                        if (receipt.status == 1) {
                            document.getElementById("txn-status").innerHTML = "Success. Seen your txn <a href='https://"+_NETWORK+"etherscan.io/tx/"+receipt.transactionHash+"'>HERE</a>";
                            document.getElementById("result").style.display = "block";
                        }
                        if (receipt.status == 0) {
                            console.log(receipt.transactionHash);
                            alert("Error minting");
                        }
                        document.getElementById("minter").classList.remove("minting");
                    });
                }).catch(function(err) {
                    document.getElementById("minter").classList.remove("minting");

                    // they cancelled.
                    if (err.code == 4001) {
                        return;
                    }

                    console.log(err);
                    alert("Error occured. Check console.");
                    return;
                });

            });
        });
